    1. // States
let todos = [];
let navState = 'all';
const $nav = document.querySelector('.nav');
const $inputTodo = document.querySelector('.input-todo');
const $todos = document.querySelector('.todos');
const $completedTodos = document.querySelector('.completed-todos');
const $activeTodos = document.querySelector('.active-todos');
const $completedAll = document.querySelector('#ck-complete-all');
const $clearCompleted = document.querySelector('.clear-completed');

// 새로 들어온 아이디에 원래 있던 아이디의 +1해주는 함수
const generateNextid = () => (todos.length ? Math.max(...todos.map(todo => todo.id)) + 1 : 1);

// html 에 뿌려주는 함수
const render = () => {
  let html = '';
  const _todos = todos.filter(todo => (navState === 'completed' ? todo.completed : navState === 'active' ? !todo.completed : true));
  _todos.forEach(({ id, content, completed }) => {
    html += `<li id="${id}" class="todo-item">
      <input id="ck-${id}" class="checkbox" type="checkbox" ${completed ? 'checked' : ''}>
      <label for="ck-${id}">${content}</label>
      <i class="remove-todo far fa-times-circle"></i>
    </li>`;
  });
  $todos.innerHTML = html;
  $completedTodos.textContent = todos.filter(todo => todo.completed).length;
  $activeTodos.textContent = todos.filter(todo => !todo.completed).length;
};

// 데이터를 서버에서 받아와 랜더링 시켜줄 함수
const fetchTodos = () => {
  fetch('/todos')
    // parse
    .then(res => res.json())
    // resolve의 response인 _todos를 빈 객체인 todos에 할당해준다.
    .then(_todos => { todos = _todos; })
    // todos를 받아왔으니 render해준다.
    .then(render)
    // 에러 처리
    .catch(console.error);
};

// 돔 + 리소스가 업데이트 될때마다 호출
window.onload = fetchTodos();

// input에 keyup이벤트 발생시 새로운 todo 추가
$inputTodo.onkeyup = e => {
  if (e.key !== 'Enter') return;
  console.log(e.target);
  const newTodo = { id: generateNextid(), content: e.target.value, completed: false };
  fetch('/todos', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(newTodo)
  })
    .then(res => res.json())
    .then(_todos => { todos = _todos; })
    .then(render)
    .catch(console.error);
};

// todos에 remove-todo 버튼 클릭 시 선택한 요소 제거
$todos.onclick = ({ target }) => {
  if (!target.matches('.todos .remove-todo')) return;
  const { id } = target.parentNode;
  fetch(`/todos/${id}`, {
    method: 'DELETE'
  })
    .then(res => res.json())
    .then(_todos => { todos = _todos; })
    .then(render)
    .catch(console.error);
};

// 체크 박스 실시간 반영
$todos.onchange = ({ target }) => {
  const { id } = target.parentNode;
  fetch(`/todos/${id}`, {
    method: 'PATCH',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ completed: target.checked })
  })
    .then(res => res.json())
    .then(_todos => { todos = _todos; })
    .then(render)
    .catch(console.error);
};

// 모든 할일의 completed를 일괄 변경
$completedAll.onchange = ({ target }) => {
  fetch('/todos/completed', {
    method: 'PATCH',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ completed: target.checked })
  })
    .then(res => res.json())
    .then(_todos => { todos = _todos; })
    .then(render)
    .catch(console.error);
};

// clear completd 버튼 누를 시 completed가 true인 요소 삭제
$clearCompleted.onclick = () => {
  fetch('/todos/completed', {
    method: 'DELETE'
  })
    .then(res => res.json())
    .then(_todos => { todos = _todos; })
    .then(render)
    .catch(console.error);
};

// nav > li 누를 시 눌러진 요소는 active 클래스 추가, 안눌려진 요소는 active 클래스 제거 
$nav.onclick = ({ target }) => {
  if (!target.matches('.nav > li')) return;
  // 모든 요소에 active를 뗀다음
  document.querySelector('.active').classList.remove('active');
  // 선택된 요소만 붙인다 
  target.classList.add('active');
  navState = target.id;
  render();
};
